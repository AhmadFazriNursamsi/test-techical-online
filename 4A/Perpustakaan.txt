  CREATE DATABASE arkademy;

  CREATE TABLE category(
    id INT AUTO_INCREMENT PRIMARY KEY,
    salary VARCHAR(100)
  );

  CREATE TABLE work(
    id INT AUTO_INCREMENT PRIMARY KEY,
    work VARCHAR(100),
    id_salary INT,
    FOREIGN KEY (id_salary) REFERENCES category(id)
  );

  CREATE TABLE name(
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    id_work INT,
    id_salary INT,
    FOREIGN KEY (id_work) REFERENCES work(id),
    FOREIGN KEY (id_salary) REFERENCES category(id)
  );
